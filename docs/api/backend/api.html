  72def _load_agent_module(agent_name: str, module_file: str = "graph.py"):
  73    """
  74    Load an agent module with proper isolation to avoid naming conflicts.
  75
  76    Args:
  77        agent_name: Name of the agent directory (e.g., 'pulmonary_research_agent')
  78        module_file: Name of the module file to load (default: 'graph.py')
  79
  80    Returns:
  81        Loaded module
  82    """
  83    agent_dir = str(Path(__file__).parent.parent / "agents" / agent_name)
  84    module_path = Path(agent_dir) / module_file
  85
  86    # Use unique module name to avoid conflicts
  87    unique_module_name = f"{agent_name}_{module_file.replace('.py', '')}"
  88
  89    spec = importlib.util.spec_from_file_location(unique_module_name, str(module_path))
  90    if spec is None or spec.loader is None:
  91        raise RuntimeError(f"Unable to load agent module spec from {module_path}")
  92
  93    module = importlib.util.module_from_spec(spec)
  94    prior = sys.modules.get(unique_module_name)
  95    sys.modules[unique_module_name] = module
  96    original_sys_path = list(sys.path)
  97    if agent_dir not in sys.path:
  98        sys.path.insert(0, agent_dir)
  99    try:
 100        spec.loader.exec_module(module)
 101    finally:
 102        sys.path = original_sys_path
 103        if prior is None:
 104            sys.modules.pop(unique_module_name, None)
 105        else:
 106            sys.modules[unique_module_name] = prior
 107
 108    return module

 111def _trim_to_last_sentence(text: str) -> str:
 112    text = (text or "").strip()
 113    if not text:
 114        return ""
 115
 116    last_dot = text.rfind(".")
 117    last_bang = text.rfind("!")
 118    last_q = text.rfind("?")
 119    last_end = max(last_dot, last_bang, last_q)
 120
 121    if last_end == -1:
 122        return text
 123
 124    return text[: last_end + 1].strip()

 127def _redact_pii(value: Any) -> Any:
 128    keys_to_drop = {
 129        "patient_email",
 130        "patient_cell",
 131        "email",
 132        "phone",
 133        "insurance_number",
 134        "emergency_contacts",
 135    }
 136
 137    if isinstance(value, dict):
 138        out: Dict[str, Any] = {}
 139        for k, v in value.items():
 140            if str(k) in keys_to_drop:
 141                continue
 142            out[str(k)] = _redact_pii(v)
 143        return out
 144
 145    if isinstance(value, list):
 146        return [_redact_pii(v) for v in value]
 147
 148    if isinstance(value, str):
 149        s = value
 150        s = re.sub(r"[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}", "[REDACTED]", s, flags=re.I)
 151        s = re.sub(r"\b\d{3}[- .]?\d{3}[- .]?\d{4}\b", "[REDACTED]", s)
 152        return s
 153
 154    return value

 157def _strip_html_to_text(html: str) -> str:
 158    h = str(html or "")
 159    if not h.strip():
 160        return ""
 161    h = re.sub(r"(?is)<script.*?>.*?</script>", " ", h)
 162    h = re.sub(r"(?is)<style.*?>.*?</style>", " ", h)
 163    paras = re.findall(r"(?is)<p[^>]*>(.*?)</p>", h)
 164    if paras:
 165        text = "\n\n".join([re.sub(r"(?is)<.*?>", " ", p) for p in paras])
 166    else:
 167        text = re.sub(r"(?is)<.*?>", " ", h)
 168    text = re.sub(r"\s+", " ", text).strip()
 169    return text

 172def _normalize_research_papers(papers: Any) -> List[Dict[str, Any]]:
 173    if not isinstance(papers, list):
 174        return []
 175
 176    normalized: List[Dict[str, Any]] = []
 177    for p in papers:
 178        if not isinstance(p, dict):
 179            continue
 180
 181        paper = dict(p)
 182        citation = paper.get("article_citation")
 183        title = paper.get("title")
 184
 185        resolved = None
 186        try:
 187            resolved = db.get_research_paper_pmc_link(article_citation=citation, title=title)
 188        except Exception:
 189            resolved = None
 190
 191        if isinstance(resolved, str) and resolved.strip():
 192            paper["pmc_link"] = resolved.strip()
 193
 194        normalized.append(paper)
 195
 196    return normalized

 230    request_id = request.headers.get("x-request-id") or str(uuid.uuid4())
 231    request.state.request_id = request_id
 232    start = time.perf_counter()
 233    try:
 234        response = await call_next(request)
 235    except Exception:
 236        duration_ms = (time.perf_counter() - start) * 1000
 237        logger.exception(
 238            "Unhandled exception request_id=%s method=%s path=%s duration_ms=%.2f",
 239            request_id,
 240            request.method,
 241            request.url.path,
 242            duration_ms,
 243        )
 244        raise
 245
 246    try:
 247        response.headers["X-Request-ID"] = request_id
 248    except Exception:
 249        pass
 250
 251    duration_ms = (time.perf_counter() - start) * 1000
 252    logger.info(
 253        "request request_id=%s method=%s path=%s status=%s duration_ms=%.2f",
 254        request_id,
 255        request.method,
 256        request.url.path,
 257        getattr(response, "status_code", "?"),
 258        duration_ms,
 259    )
 260    return response

 506    """
 507    Search doctor notes for a patient using semantic search.
 508
 509    Args:
 510        patient_id: The patient's ID
 511        payload: JSON with 'question' field
 512
 513    Returns:
 514        Dictionary with patient info, relevant notes, and answer
 515    """
 516    try:
 517        question = payload.get("question", "")
 518        if not question:
 519            raise HTTPException(status_code=400, detail="Question is required")
 520
 521        doctor_name = str(payload.get("doctor_name") or "Tiffany Mitchell")
 522        patient_name = str(payload.get("patient_name") or "").strip()
 523        if not patient_name:
 524            try:
 525                patient = db.get_patient(patient_id)
 526                patient_name = str((patient or {}).get("name") or "").strip()
 527            except Exception:
 528                patient_name = ""
 529
 530        timestamp = _now_utc_iso_z()
 531        question_id = f"dq_{int(datetime.now().timestamp() * 1000)}"
 532        question_doc = {
 533            "question_asked": question,
 534            "patient_name": patient_name,
 535            "doctor_name": doctor_name,
 536            "timestamp": timestamp,
 537        }
 538        try:
 539            db.save_doctors_question(question_id, question_doc)
 540        except Exception as e:
 541            logger.warning("Warning: Failed to save doctors question: %s", e)
 542
 543        logger.info(
 544            "search_patient_doctor_notes patient_id=%s question_len=%s patient_name=%s",
 545            patient_id,
 546            len(question),
 547            patient_name,
 548        )
 549
 550        # Use the doc notes search agent directly
 551        state = DocNotesSearcher.build_starting_state(patient_id=patient_id, question=question)
 552        if patient_name:
 553            state["patient_name"] = patient_name
 554
 555        # Add the question as a human message in JSON format
 556        state["messages"].append(
 557            langchain_core.messages.HumanMessage(
 558                content=f'{{"patient_id": "{patient_id}", "patient_name": "{patient_name or ""}", "question": "{question}"}}'
 559            )
 560        )
 561
 562        # Invoke the agent
 563        logger.info("Invoking DocNotesSearcher agent for patient_id=%s", patient_id)
 564        request_id = getattr(getattr(request, "state", None), "request_id", None)
 565        search_span = _root_span.new(
 566            name="DocNotesSearcher.invoke",
 567            agent="docnotes_search_agent",
